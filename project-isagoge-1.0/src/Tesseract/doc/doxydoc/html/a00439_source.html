62 // Helper to remove an enclosing circle from an image.
63 // If there isn't one, then the image will most likely get badly mangled.
64 // The returned pix must be pixDestroyed after use. NULL may be returned
65 // if the image doesn't meet the trivial conditions that it uses to determine
66 // success.
67 static Pix* RemoveEnclosingCircle(Pix* pixs) {
68  Pix* pixsi = pixInvert(NULL, pixs);
69  Pix* pixc = pixCreateTemplate(pixs);
70  pixSetOrClearBorder(pixc, 1, 1, 1, 1, PIX_SET);
71  pixSeedfillBinary(pixc, pixc, pixsi, 4);
72  pixInvert(pixc, pixc);
73  pixDestroy(&pixsi);
74  Pix* pixt = pixAnd(NULL, pixs, pixc);
75  l_int32 max_count;
76  pixCountConnComp(pixt, 8, &max_count);
77  // The count has to go up before we start looking for the minimum.
78  l_int32 min_count = MAX_INT32;
79  Pix* pixout = NULL;
80  for (int i = 1; i < kMaxCircleErosions; i++) {
81  pixDestroy(&pixt);
82  pixErodeBrick(pixc, pixc, 3, 3);
83  pixt = pixAnd(NULL, pixs, pixc);
84  l_int32 count;
85  pixCountConnComp(pixt, 8, &count);
86  if (i == 1 || count > max_count) {
87  max_count = count;
88  min_count = count;
89  } else if (i > 1 && count < min_count) {
90  min_count = count;
91  pixDestroy(&pixout);
92  pixout = pixCopy(NULL, pixt); // Save the best.
93  } else if (count >= min_count) {
94  break; // We have passed by the best.
95  }
96  }
97  pixDestroy(&pixt);
98  pixDestroy(&pixc);
99  return pixout;
100 }

107 int Tesseract::SegmentPage(const STRING* input_file, BLOCK_LIST* blocks,
108  Tesseract* osd_tess, OSResults* osr) {
109  ASSERT_HOST(pix_binary_ != NULL);
110  int width = pixGetWidth(pix_binary_);
111  int height = pixGetHeight(pix_binary_);
112  // Get page segmentation mode.
113  PageSegMode pageseg_mode = static_cast<PageSegMode>(
114  static_cast<int>(tessedit_pageseg_mode));
115  // If a UNLV zone file can be found, use that instead of segmentation.
116  if (!PSM_COL_FIND_ENABLED(pageseg_mode) &&
117  input_file != NULL && input_file->length() > 0) {
118  STRING name = *input_file;
119  const char* lastdot = strrchr(name.string(), '.');
120  if (lastdot != NULL)
121  name[lastdot - name.string()] = '\0';
122  read_unlv_file(name, width, height, blocks);
123  }
124  if (blocks->empty()) {
125  // No UNLV file present. Work according to the PageSegMode.
126  // First make a single block covering the whole image.
127  BLOCK_IT block_it(blocks);
128  BLOCK* block = new BLOCK("", TRUE, 0, 0, 0, 0, width, height);
129  block->set_right_to_left(right_to_left());
130  block_it.add_to_end(block);
131  } else {
132  // UNLV file present. Use PSM_SINGLE_BLOCK.
133  pageseg_mode = PSM_SINGLE_BLOCK;
134  }
135  int auto_page_seg_ret_val = 0;
136  TO_BLOCK_LIST to_blocks;
137  if (PSM_OSD_ENABLED(pageseg_mode) || PSM_BLOCK_FIND_ENABLED(pageseg_mode) ||
138  PSM_SPARSE(pageseg_mode)) {
139  auto_page_seg_ret_val =
140  AutoPageSeg(pageseg_mode, blocks, &to_blocks, osd_tess, osr);
141  if (pageseg_mode == PSM_OSD_ONLY)
142  return auto_page_seg_ret_val;
143  // To create blobs from the image region bounds uncomment this line:
144  // to_blocks.clear(); // Uncomment to go back to the old mode.
145  } else {
146  deskew_ = FCOORD(1.0f, 0.0f);
147  reskew_ = FCOORD(1.0f, 0.0f);
148  if (pageseg_mode == PSM_CIRCLE_WORD) {
149  Pix* pixcleaned = RemoveEnclosingCircle(pix_binary_);
150  if (pixcleaned != NULL) {
151  pixDestroy(&pix_binary_);
152  pix_binary_ = pixcleaned;
153  }
154  }
155  }
156 
157  if (auto_page_seg_ret_val < 0) {
158  return -1;
159  }
160 
161  if (blocks->empty()) {
162  if (textord_debug_tabfind)
163  tprintf("Empty page\n");
164  return 0; // AutoPageSeg found an empty page.
165  }
166 
167  textord_.TextordPage(pageseg_mode, width, height, pix_binary_,
168  blocks, &to_blocks);
169  return auto_page_seg_ret_val;
170 }

172 // Helper writes a grey image to a file for use by scrollviewer.
173 // Normally for speed we don't display the image in the layout debug windows.
174 // If textord_debug_images is true, we draw the image as a background to some
175 // of the debug windows. printable determines whether these
176 // images are optimized for printing instead of screen display.
177 static void WriteDebugBackgroundImage(bool printable, Pix* pix_binary) {
178  Pix* grey_pix = pixCreate(pixGetWidth(pix_binary),
179  pixGetHeight(pix_binary), 8);
180  // Printable images are light grey on white, but for screen display
181  // they are black on dark grey so the other colors show up well.
182  if (printable) {
183  pixSetAll(grey_pix);
184  pixSetMasked(grey_pix, pix_binary, 192);
185  } else {
186  pixSetAllArbitrary(grey_pix, 64);
187  pixSetMasked(grey_pix, pix_binary, 0);
188  }
189  AlignedBlob::IncrementDebugPix();
190  pixWrite(AlignedBlob::textord_debug_pix().string(), grey_pix, IFF_PNG);
191  pixDestroy(&grey_pix);
192 }

214 int Tesseract::AutoPageSeg(PageSegMode pageseg_mode,
215  BLOCK_LIST* blocks, TO_BLOCK_LIST* to_blocks,
216  Tesseract* osd_tess, OSResults* osr) {
217  if (textord_debug_images) {
218  WriteDebugBackgroundImage(textord_debug_printable, pix_binary_);
219  }
220  Pix* photomask_pix = NULL;
221  Pix* musicmask_pix = NULL;
222  // The blocks made by the ColumnFinder. Moved to blocks before return.
223  BLOCK_LIST found_blocks;
224  TO_BLOCK_LIST temp_blocks;
225 
226  bool single_column = !PSM_COL_FIND_ENABLED(pageseg_mode);
227  bool osd_enabled = PSM_OSD_ENABLED(pageseg_mode);
228  bool osd_only = pageseg_mode == PSM_OSD_ONLY;
229  ColumnFinder* finder = SetupPageSegAndDetectOrientation(
230  single_column, osd_enabled, osd_only, blocks, osd_tess, osr,
231  &temp_blocks, &photomask_pix, &musicmask_pix);
232  int result = 0;
233  if (finder != NULL) {
234  TO_BLOCK_IT to_block_it(&temp_blocks);
235  TO_BLOCK* to_block = to_block_it.data();
236  if (musicmask_pix != NULL) {
237  // TODO(rays) pass the musicmask_pix into FindBlocks and mark music
238  // blocks separately. For now combine with photomask_pix.
239  pixOr(photomask_pix, photomask_pix, musicmask_pix);
240  }
241  if (equ_detect_) {
242  finder->SetEquationDetect(equ_detect_);
243  }
244  result = finder->FindBlocks(pageseg_mode, scaled_color_, scaled_factor_,
245  to_block, photomask_pix,
246  &found_blocks, to_blocks);
247  if (result >= 0)
248  finder->GetDeskewVectors(&deskew_, &reskew_);
249  delete finder;
250  }
251  pixDestroy(&photomask_pix);
252  pixDestroy(&musicmask_pix);
253  if (result < 0) return result;
254 
255  blocks->clear();
256  BLOCK_IT block_it(blocks);
257  // Move the found blocks to the input/output blocks.
258  block_it.add_list_after(&found_blocks);
259 
260  if (textord_debug_images) {
261  // The debug image is no longer needed so delete it.
262  unlink(AlignedBlob::textord_debug_pix().string());
263  }
264  return result;
265 }

280 ColumnFinder* Tesseract::SetupPageSegAndDetectOrientation(
281  bool single_column, bool osd, bool only_osd,
282  BLOCK_LIST* blocks, Tesseract* osd_tess, OSResults* osr,
283  TO_BLOCK_LIST* to_blocks, Pix** photo_mask_pix, Pix** music_mask_pix) {
284  int vertical_x = 0;
285  int vertical_y = 1;
286  TabVector_LIST v_lines;
287  TabVector_LIST h_lines;
288  ICOORD bleft(0, 0);
289 
290  ASSERT_HOST(pix_binary_ != NULL);
291  if (tessedit_dump_pageseg_images) {
292  pixWrite("tessinput.png", pix_binary_, IFF_PNG);
293  }
294  // Leptonica is used to find the rule/separator lines in the input.
295  LineFinder::FindAndRemoveLines(source_resolution_,
296  textord_tabfind_show_vlines, pix_binary_,
297  &vertical_x, &vertical_y, music_mask_pix,
298  &v_lines, &h_lines);
299  if (tessedit_dump_pageseg_images)
300  pixWrite("tessnolines.png", pix_binary_, IFF_PNG);
301  // Leptonica is used to find a mask of the photo regions in the input.
302  *photo_mask_pix = ImageFind::FindImages(pix_binary_);
303  if (tessedit_dump_pageseg_images)
304  pixWrite("tessnoimages.png", pix_binary_, IFF_PNG);
305  if (single_column)
306  v_lines.clear();
307 
308  // The rest of the algorithm uses the usual connected components.
309  textord_.find_components(pix_binary_, blocks, to_blocks);
310 
311  TO_BLOCK_IT to_block_it(to_blocks);
312  // There must be exactly one input block.
313  // TODO(rays) handle new textline finding with a UNLV zone file.
314  ASSERT_HOST(to_blocks->singleton());
315  TO_BLOCK* to_block = to_block_it.data();
316  TBOX blkbox = to_block->block->bounding_box();
317  ColumnFinder* finder = NULL;
318 
319  if (to_block->line_size >= 2) {
320  finder = new ColumnFinder(static_cast<int>(to_block->line_size),
321  blkbox.botleft(), blkbox.topright(),
322  source_resolution_,
323  &v_lines, &h_lines, vertical_x, vertical_y);
324 
325  finder->SetupAndFilterNoise(*photo_mask_pix, to_block);
326 
327  if (equ_detect_) {
328  equ_detect_->LabelSpecialText(to_block);
329  }
330 
331  BLOBNBOX_CLIST osd_blobs;
332  // osd_orientation is the number of 90 degree rotations to make the
333  // characters upright. (See osdetect.h for precise definition.)
334  // We want the text lines horizontal, (vertical text indicates vertical
335  // textlines) which may conflict (eg vertically written CJK).
336  int osd_orientation = 0;
337  bool vertical_text = finder->IsVerticallyAlignedText(to_block, &osd_blobs);
338  if (osd && osd_tess != NULL && osr != NULL) {
339  os_detect_blobs(&osd_blobs, osr, osd_tess);
340  if (only_osd) {
341  delete finder;
342  return NULL;
343  }
344  osd_orientation = osr->best_result.orientation_id;
345  double osd_score = osr->orientations[osd_orientation];
346  double osd_margin = min_orientation_margin * 2;
347  for (int i = 0; i < 4; ++i) {
348  if (i != osd_orientation &&
349  osd_score - osr->orientations[i] < osd_margin) {
350  osd_margin = osd_score - osr->orientations[i];
351  }
352  }
353  if (osd_margin < min_orientation_margin) {
354  // The margin is weak.
355  int best_script_id = osr->best_result.script_id;
356  bool cjk = (best_script_id == osd_tess->unicharset.han_sid()) ||
357  (best_script_id == osd_tess->unicharset.hiragana_sid()) ||
358  (best_script_id == osd_tess->unicharset.katakana_sid());
359 
360  if (!cjk && !vertical_text && osd_orientation == 2) {
361  // upside down latin text is improbable with such a weak margin.
362  tprintf("OSD: Weak margin (%.2f), horiz textlines, not CJK: "
363  "Don't rotate.\n", osd_margin);
364  osd_orientation = 0;
365  } else {
366  tprintf("OSD: Weak margin (%.2f) for %d blob text block, "
367  "but using orientation anyway: %d\n",
368  osd_blobs.length(), osd_margin, osd_orientation);
369  }
370  }
371  }
372  osd_blobs.shallow_clear();
373  finder->CorrectOrientation(to_block, vertical_text, osd_orientation);
374  }
375 
376  return finder;
377 }
