339 * whitespace spaced string being an arg.

342 int
343 parse_args ( /*refine arg list */
344  int argc, /*no of input args */
345  char *argv[], /*input args */
346  char *arglist[] /*output args */
347  ) {
348  int argcount; /*converted argc */
349  char *testchar; /*char in option string */
350  int arg; /*current argument */
351 
352  argcount = 0; /*no of options */
353  for (arg = 0; arg < argc; arg++) {
354  testchar = argv[arg]; /*start of arg */
355  do {
356  while (*testchar
357  && (*testchar == ' ' || *testchar == '\n'
358  || *testchar == '\t'))
359  testchar++; /*skip white space */
360  if (*testchar) {
361  /*new arg */
362  arglist[argcount++] = testchar;
363  /*skip to white space */
364  for (testchar++; *testchar && *testchar != ' ' && *testchar != '\n' && *testchar != '\t'; testchar++) ;
365  if (*testchar)
366  *testchar++ = '\0'; /*turn to separate args */
367  }
368  }
369  while (*testchar);
370  }
371  return argcount; /*new number of args */
372 }
