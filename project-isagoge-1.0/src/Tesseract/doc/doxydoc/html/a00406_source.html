213 int TessBaseAPI::Init(const char* datapath, const char* language,
214  OcrEngineMode oem, char **configs, int configs_size,
215  const GenericVector<STRING> *vars_vec,
216  const GenericVector<STRING> *vars_values,
217  bool set_only_non_debug_params) {
218  // Default language is "eng".
219  if (language == NULL) language = "eng";
220  // If the datapath, OcrEngineMode or the language have changed - start again.
221  // Note that the language_ field stores the last requested language that was
222  // initialized successfully, while tesseract_->lang stores the language
223  // actually used. They differ only if the requested language was NULL, in
224  // which case tesseract_->lang is set to the Tesseract default ("eng").
225  if (tesseract_ != NULL &&
226  (datapath_ == NULL || language_ == NULL ||
227  *datapath_ != datapath || last_oem_requested_ != oem ||
228  (*language_ != language && tesseract_->lang != language))) {
229  delete tesseract_;
230  tesseract_ = NULL;
231  }
232 
233  bool reset_classifier = true;
234  if (tesseract_ == NULL) {
235  reset_classifier = false;
236  tesseract_ = new Tesseract;
237  if (tesseract_->init_tesseract(
238  datapath, output_file_ != NULL ? output_file_->string() : NULL,
239  language, oem, configs, configs_size, vars_vec, vars_values,
240  set_only_non_debug_params) != 0) {
241  return -1;
242  }
243  }
244  // Update datapath and language requested for the last valid initialization.
245  if (datapath_ == NULL)
246  datapath_ = new STRING(datapath);
247  else
248  *datapath_ = datapath;
249  if (language_ == NULL)
250  language_ = new STRING(language);
251  else
252  *language_ = language;
253  last_oem_requested_ = oem;
254 
255  // For same language and datapath, just reset the adaptive classifier.
256  if (reset_classifier) tesseract_->ResetAdaptiveClassifier();
257 
258  return 0;
259 }

556 Boxa* TessBaseAPI::GetComponentImages(PageIteratorLevel level,
557  bool text_only,
558  Pixa** pixa, int** blockids) {
559  PageIterator* page_it = GetIterator();
560  if (page_it == NULL)
561  page_it = AnalyseLayout();
562  if (page_it == NULL)
563  return NULL; // Failed.
564 
565  // Count the components to get a size for the arrays.
566  int component_count = 0;
567  int left, top, right, bottom;
568  do {
569  if (page_it->BoundingBoxInternal(level, &left, &top, &right, &bottom) &&
570  (!text_only || PTIsTextType(page_it->BlockType())))
571  ++component_count;
572  } while (page_it->Next(level));
573 
574  Boxa* boxa = boxaCreate(component_count);
575  if (pixa != NULL)
576  *pixa = pixaCreate(component_count);
577  if (blockids != NULL)
578  *blockids = new int[component_count];
579 
580  int blockid = 0;
581  int component_index = 0;
582  page_it->Begin();
583  do {
584  if (page_it->BoundingBoxInternal(level, &left, &top, &right, &bottom) &&
585  (!text_only || PTIsTextType(page_it->BlockType()))) {
586  Box* lbox = boxCreate(left, top, right - left, bottom - top);
587  boxaAddBox(boxa, lbox, L_INSERT);
588  if (pixa != NULL) {
589  Pix* pix = page_it->GetBinaryImage(level);
590  pixaAddPix(*pixa, pix, L_INSERT);
591  pixaAddBox(*pixa, lbox, L_CLONE);
592  }
593  if (blockids != NULL) {
594  (*blockids)[component_index] = blockid;
595  if (page_it->IsAtFinalElement(RIL_BLOCK, level))
596  ++blockid;
597  }
598  ++component_index;
599  }
600  } while (page_it->Next(level));
601  delete page_it;
602  return boxa;
603 }

705 int TessBaseAPI::Recognize(ETEXT_DESC* monitor) {
706  if (tesseract_ == NULL)
707  return -1;
708  if (FindLines() != 0)
709  return -1;
710  if (page_res_ != NULL)
711  delete page_res_;
712  if (block_list_->empty()) {
713  page_res_ = new PAGE_RES(block_list_, &tesseract_->prev_word_best_choice_);
714  return 0; // Empty page.
715  }
716 
717  tesseract_->SetBlackAndWhitelist();
718  recognition_done_ = true;
719  if (tesseract_->tessedit_resegment_from_line_boxes)
720  page_res_ = tesseract_->ApplyBoxes(*input_file_, true, block_list_);
721  else if (tesseract_->tessedit_resegment_from_boxes)
722  page_res_ = tesseract_->ApplyBoxes(*input_file_, false, block_list_);
723  else
724  page_res_ = new PAGE_RES(block_list_, &tesseract_->prev_word_best_choice_);
725  if (tesseract_->tessedit_make_boxes_from_boxes) {
726  tesseract_->CorrectClassifyWords(page_res_);
727  return 0;
728  }
729 
730  if (truth_cb_ != NULL) {
731  tesseract_->wordrec_run_blamer.set_value(true);
732  truth_cb_->Run(tesseract_->getDict().getUnicharset(),
733  image_height_, page_res_);
734  }
735 
736  int result = 0;
737  if (tesseract_->interactive_display_mode) {
738  #ifndef GRAPHICS_DISABLED
739  tesseract_->pgeditor_main(rect_width_, rect_height_, page_res_);
740  #endif // GRAPHICS_DISABLED
741  // The page_res is invalid after an interactive session, so cleanup
742  // in a way that lets us continue to the next page without crashing.
743  delete page_res_;
744  page_res_ = NULL;
745  return -1;
746  } else if (tesseract_->tessedit_train_from_boxes) {
747  tesseract_->ApplyBoxTraining(*output_file_, page_res_);
748  } else if (tesseract_->tessedit_ambigs_training) {
749  FILE *training_output_file = tesseract_->init_recog_training(*input_file_);
750  // OCR the page segmented into words by tesseract.
751  tesseract_->recog_training_segmented(
752  *input_file_, page_res_, monitor, training_output_file);
753  fclose(training_output_file);
754  } else {
755  // Now run the main recognition.
756  if (tesseract_->recog_all_words(page_res_, monitor, NULL, NULL, 0)) {
757  DetectParagraphs(true);
758  } else {
759  result = -1;
760  }
761  }
762  return result;
763 }
